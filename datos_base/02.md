## Impacto y riesgos

Qué puede romperse:

* hoy el renombrado depende de reglas fijas dentro de `datos_base/renombrar_pdf.py`; si se cambia mal, se puede dejar de renombrar `002.pdf`, `ORS.pdf` o `PI.pdf` como hoy lo hace el script actual 
* una regla Oracle demasiado amplia puede renombrar muchos PDFs de forma incorrecta
* si el destino ya existe, puede haber colisión
* si el PDF no trae texto seleccionable, la validación por contenido no va a encontrar coincidencia

Qué se va a proteger:

* prioridad fija: primero contenido, luego nombre
* no sobrescribir archivos existentes
* modo simulación
* logs claros del motivo del cambio
* fallback seguro: si no encuentra regla, no cambia nada

```python
import os
import re
import fitz
import unicodedata
from dataclasses import dataclass

# ============================================================
# NUEVO: conexión Oracle centralizada para leer reglas
# Ajustar estas variables si el entorno ya las maneja por otro lado
# ============================================================
try:
    import oracledb
except ImportError:
    oracledb = None

ORACLE_USER = os.getenv("ORACLE_USER", "ADMINISTRATIVO")
ORACLE_PASSWORD = os.getenv("ORACLE_PASSWORD", "")
ORACLE_DSN = os.getenv("ORACLE_DSN", "")  # ejemplo: "10.10.10.10:1521/ORCL"

print("=== RENOMBRADO FINAL DESDE ORACLE ===\n")


@dataclass
class ReglaPdf:
    nombre_pdf: str
    activo: str
    orden: int
    nota: str | None
    regla_similaridad: str | None
    regla_lee_documento: str | None


def normalizar_texto(valor: str | None) -> str:
    if not valor:
        return ""

    valor = unicodedata.normalize("NFKD", valor)
    valor = "".join(ch for ch in valor if not unicodedata.combining(ch))
    valor = valor.upper()
    valor = re.sub(r"\s+", " ", valor).strip()
    return valor


def normalizar_compacto(valor: str | None) -> str:
    valor = normalizar_texto(valor)
    valor = re.sub(r"[^A-Z0-9]+", "", valor)
    return valor


def obtener_conexion_oracle():
    if oracledb is None:
        raise RuntimeError(
            "No está instalado oracledb. Ejecutar: pip install oracledb"
        )

    if not ORACLE_USER or not ORACLE_PASSWORD or not ORACLE_DSN:
        raise RuntimeError(
            "Faltan variables ORACLE_USER / ORACLE_PASSWORD / ORACLE_DSN"
        )

    return oracledb.connect(
        user=ORACLE_USER,
        password=ORACLE_PASSWORD,
        dsn=ORACLE_DSN,
    )


def cargar_reglas_desde_oracle() -> list[ReglaPdf]:
    conn = obtener_conexion_oracle()
    try:
        cur = conn.cursor()
        cur.execute(
            """
            SELECT
                NOMBRE_PDF,
                ACTIVO,
                NVL(ORDEN, 999999) AS ORDEN,
                NOTA,
                REGLA_SIMILARIDAD,
                REGLA_LEE_DOCUMENTO
            FROM PDF_NOMBRES_VALIDOS
            WHERE ACTIVO = 'S'
            ORDER BY NVL(ORDEN, 999999), NOMBRE_PDF
            """
        )

        reglas = []
        for row in cur.fetchall():
            reglas.append(
                ReglaPdf(
                    nombre_pdf=row[0],
                    activo=row[1],
                    orden=int(row[2]) if row[2] is not None else 999999,
                    nota=row[3],
                    regla_similaridad=row[4],
                    regla_lee_documento=row[5],
                )
            )
        return reglas
    finally:
        conn.close()


def extraer_texto_pdf(path_pdf: str) -> str:
    partes = []
    doc = fitz.open(path_pdf)
    try:
        for page in doc:
            try:
                partes.append(page.get_text("text") or "")
            except Exception:
                continue
    finally:
        doc.close()

    return "\n".join(partes)


def resolver_nombre_desde_reglas(nombre_archivo: str, texto_pdf: str, reglas: list[ReglaPdf]):
    nombre_normal = normalizar_texto(nombre_archivo)
    nombre_compacto = normalizar_compacto(nombre_archivo)

    texto_normal = normalizar_texto(texto_pdf)
    texto_compacto = normalizar_compacto(texto_pdf)

    # ============================================================
    # PRIORIDAD 1: buscar coincidencia por contenido del PDF
    # ============================================================
    for regla in reglas:
        patron = (regla.regla_lee_documento or "").strip()
        if not patron:
            continue

        patron_normal = normalizar_texto(patron)
        patron_compacto = normalizar_compacto(patron)

        if patron_normal and patron_normal in texto_normal:
            return regla.nombre_pdf, f"contenido:{patron}"

        if patron_compacto and patron_compacto in texto_compacto:
            return regla.nombre_pdf, f"contenido_compacto:{patron}"

    # ============================================================
    # PRIORIDAD 2: buscar coincidencia por nombre del archivo
    # ============================================================
    for regla in reglas:
        patron = (regla.regla_similaridad or "").strip()
        if not patron:
            continue

        patron_normal = normalizar_texto(patron)
        patron_compacto = normalizar_compacto(patron)

        if patron_normal and patron_normal in nombre_normal:
            return regla.nombre_pdf, f"nombre:{patron}"

        if patron_compacto and patron_compacto in nombre_compacto:
            return regla.nombre_pdf, f"nombre_compacto:{patron}"

    return None, "sin_coincidencia"


def renombrar_pdfs(base_dir: str, dry_run: bool = True):
    reglas = cargar_reglas_desde_oracle()

    print(f"Reglas activas cargadas desde Oracle: {len(reglas)}")
    print(f"Modo simulación: {'SI' if dry_run else 'NO'}\n")

    total = 0
    simulados = 0
    renombrados = 0
    sin_cambios = 0
    colisiones = 0
    errores = 0

    for root, dirs, files in os.walk(base_dir):
        for file in files:
            if not file.lower().endswith(".pdf"):
                continue

            total += 1
            path = os.path.join(root, file)

            try:
                texto_pdf = extraer_texto_pdf(path)
                nombre_destino, motivo = resolver_nombre_desde_reglas(file, texto_pdf, reglas)

                if not nombre_destino:
                    sin_cambios += 1
                    print(f" - Sin cambios: {path} | motivo={motivo}")
                    continue

                new_path = os.path.join(root, nombre_destino)

                if os.path.abspath(path) == os.path.abspath(new_path):
                    sin_cambios += 1
                    print(f" - Ya correcto: {path} | motivo={motivo}")
                    continue

                if os.path.exists(new_path):
                    colisiones += 1
                    print(f"⚠ Colisión: {path} -> {new_path} | motivo={motivo}")
                    continue

                if dry_run:
                    simulados += 1
                    print(f"✓ Simulado: {path} -> {new_path} | motivo={motivo}")
                else:
                    os.rename(path, new_path)
                    renombrados += 1
                    print(f"✓ Renombrado: {path} -> {new_path} | motivo={motivo}")

            except Exception as e:
                errores += 1
                print(f"X Error: {path} | detalle={e}")

    print("\n=== RESUMEN ===")
    print("Total PDFs:", total)
    print("Simulados:", simulados)
    print("Renombrados:", renombrados)
    print("Sin cambios:", sin_cambios)
    print("Colisiones:", colisiones)
    print("Errores:", errores)
    print("\nTERMINADO")


if __name__ == "__main__":
    base_dir = os.getcwd()

    # ============================================================
    # NUEVO: primero probar en simulación
    # Cambiar a False cuando ya se haya validado
    # ============================================================
    renombrar_pdfs(base_dir=base_dir, dry_run=True)
```

## 3) SQL de control recomendado en Oracle

Para revisar reglas activas:

```sql
SELECT NOMBRE_PDF,
       ACTIVO,
       ORDEN,
       NOTA,
       REGLA_SIMILARIDAD,
       REGLA_LEE_DOCUMENTO
FROM PDF_NOMBRES_VALIDOS
WHERE ACTIVO = 'S'
ORDER BY ORDEN, NOMBRE_PDF;
````

Para encontrar duplicadas por similitud:

```sql
SELECT REGLA_SIMILARIDAD, COUNT(*)
FROM PDF_NOMBRES_VALIDOS
WHERE ACTIVO = 'S'
  AND REGLA_SIMILARIDAD IS NOT NULL
GROUP BY REGLA_SIMILARIDAD
HAVING COUNT(*) > 1;
```

Para encontrar duplicadas por contenido:

```sql
SELECT REGLA_LEE_DOCUMENTO, COUNT(*)
FROM PDF_NOMBRES_VALIDOS
WHERE ACTIVO = 'S'
  AND REGLA_LEE_DOCUMENTO IS NOT NULL
GROUP BY REGLA_LEE_DOCUMENTO
HAVING COUNT(*) > 1;
```

## Pruebas de verificación y regresión

### Verificación nueva

1. Un PDF cuyo contenido tenga `HOJA DE COBERTURA` debe resolver a `CC.pdf` porque esa regla ya existe en la tabla 

2. Un PDF cuyo nombre contenga el patrón configurado en `REGLA_SIMILARIDAD` debe renombrarse al `NOMBRE_PDF` correspondiente.

3. Un PDF sin coincidencia no debe cambiarse.

4. Si el archivo ya se llama como el destino, debe salir como “Ya correcto”.

5. Si el destino ya existe, debe salir “Colisión”.

### Regresión de no-rotura

Debe seguir funcionando:

* lectura de texto por `fitz`, que ya usa el script actual 
* recorrido recursivo por carpetas
* renombrado sólo de `.pdf`
* ejecución desde cliente escritorio sin web
* operación masiva sin tener que distribuir reglas por equipo

Con la data que se ve en la tabla, `PI.pdf` tiene `REGLA_SIMILARIDAD = 'F_NOMBRE'`, no `PLANILLA`
